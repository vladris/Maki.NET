    Class ActionExtensions

    Provides Action extension methods for converting any Action to an equivalent Func which returns a Unit. With such a conversion, Action and Func can be treated uniformly.

    Examples
    The following example shows how ToFunc() adapter can be use to get a Func<Unit> from an Action. 
    using Maki.Functional;
using System;

namespace Samples
{
    class Program
    {
        // Generic method expects a Func argument
        public static T Invoke<T>(Func<T> func)
        {
            return func();
        }

        static void Main(string[] args)
        {
            // An action does not return a value
            Action act = () => 
            {
                Console.WriteLine("Hello world!");
            };

            // ToFunc adapter converts an Action to a Func<Unit>
            Invoke(act.ToFunc());
        }
    }
}
